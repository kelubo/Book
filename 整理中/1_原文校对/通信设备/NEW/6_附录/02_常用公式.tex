\section{频率与波长}

\begin{equation}
    c = f \times \lambda
    \label{eq:wave_speed}
\end{equation}

其中：
- \( c \) 为光速（约 \( 3 \times 10^8 \, \text{m/s} \)）
- \( f \) 为频率（单位：Hz）
- \( \lambda \) 为波长（单位：m）

\section{谐振与调谐}

\begin{equation}
    f_0 = \frac{1}{2\pi\sqrt{LC}}
    \label{eq:resonant_frequency}
\end{equation}

其中：
- \( f_0 \) 为谐振频率（单位：Hz）
- \( L \) 为电感（单位：H）
- \( C \) 为电容（单位：F）

\section{功率与增益}

\begin{equation}
    \text{增益 (dB)} = 10 \log_{10}\left(\frac{P_{\text{out}}}{P_{\text{in}}}\right)
    \label{eq:gain_dB}
\end{equation}

\begin{equation}
    \text{功率 (dBm)} = 10 \log_{10}\left(\frac{P}{1 \, \text{mW}}\right)
    \label{eq:power_dBm}
\end{equation}

\section{调制与解调}

超外差接收机的中频频率：

\begin{equation}
    f_{\text{IF}} = |f_{\text{LO}} - f_{\text{RF}}|
    \label{eq:IF_frequency}
\end{equation}

其中：
- \( f_{\text{IF}} \) 为中频频率
- \( f_{\text{LO}} \) 为本振频率
- \( f_{\text{RF}} \) 为射频频率
